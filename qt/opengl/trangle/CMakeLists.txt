cmake_minimum_required(VERSION 3.22)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOUIC_SEARCH_PATHS uis)
find_package(Qt6 COMPONENTS Core Gui Widgets OpenGLWidgets REQUIRED)
qt_add_executable(trangle main.cpp mainwindow.cpp mywidget.cpp uis/mainwindow.ui)
qt_add_resources(trangle "trangle" PREFIX "/" FILES res.qrc)
target_link_libraries(trangle PUBLIC Qt::Core Qt::Gui Qt::Widgets Qt::OpenGLWidgets)
